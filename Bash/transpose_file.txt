# Read from the file file.txt and print its transposed content to stdout.
awk '
{
    for(i=1;i<=NF;i++)
    {
        if(FNR==1)
        {
            t[i]=$i;
        }
        else
        {
            t[i]=t[i] " " $i;
        }
    }
}
END{
    for(i=1;t[i]!="";i++)
    {
        print t[i];
    }
}
' file.txt
